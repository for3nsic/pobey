%{
#include<stdio.h>
int vowels=0;
int cons=0;
%}
%%
[aeiouAEIOU] {vowels++;}
[a-zA-Z] {cons++;}
%%
int yywrap()
{
return 1;
}
main()
{
printf(“Enter the string.. at end press ^d\n”);
yylex();
printf(“No of vowels=%d\nNo of consonants=%d\n”, vowels,cons);
}

How to Compile:
1.    Save the file with .l extention. E.g. vowels.l
2.    lex vowels.l
3.    gcc –o objfile lex.yy.c
4.    ./objfile
